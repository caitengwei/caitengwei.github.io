class ProductOfDigits
{
public:
int smallestNumber(int N)
{
int i, ret = 0;
bool f = 0;

if(N == 1 || N == 0)
return 1;
while(1)
{
f = 0;
for(i = 9; i > 1; i--)
{
if(N % i == 0)
{
N /= i;
f = 1;
ret ++;
break;
}
}
if(!f || N == 1)
{
if(N < 10)
return ret;
else
return -1;
}
}
}
};
