网络流题目
USACO Drainage Ditches

二分匹配可以转化为网络流来做，或者说，二分匹配就是特殊的网络流

附ditch代码 :
#include
#include
using namespace std;

#define SZ 201
#define MAX 100010

class edge {
int nt[SZ], sz;
public:
inline int size() {
return sz;
}

inline void clear() {
sz = 0;
}

inline void push(int x) {
nt[sz ++] = x;
}

inline int operator[] (const int &x) {
return nt[x];
}
};
edge arc[SZ];
int qq[MAX];
int n, m, ma[SZ][SZ];

int edmonds(int st, int ed) {
int fa[SZ], flow[SZ];
int ret = 0, lt = 0, rt = 0;
int cur, i, n, np, tmp;
memset(fa, 0, sizeof(fa));
qq[rt++] = st;
flow[st] = 0x7fffffff;
while(lt != rt) {
cur = qq[lt ++];
if(lt == MAX) lt = 0;
n = arc[cur].size();
for(i = 0; i < n; i++) {
np = arc[cur][i];
if(!ma[cur][np] || np == st) continue;
if(fa[np]) continue;
fa[np] = cur;
flow[np] = min(ma[cur][np], flow[cur]);
qq[rt++] = np;
if(rt == MAX) rt = 0;
if(np == ed) {
tmp = flow[np];
while(np != st) {
ma[fa[np]][np] -= tmp;
if(!ma[np][fa[np]]) {
arc[np].push(fa[np]);
ma[np][fa[np]] = tmp;
}
else
ma[np][fa[np]] += tmp;
np = fa[np];
}
memset(fa, 0, sizeof(fa));
flow[st] = 0x7fffffff;
ret += tmp;
lt = 0, rt = 0;
qq[rt ++] = st;
break;
}
}
}

return ret;
}

int main()
{
int n, m;
while(cin >> m >> n) {
int i, j, a, b, c;
memset(ma, 0, sizeof(ma));
for(i = 1; i <= n; i++) {
arc[i].clear();
}
for(i = 0; i < m; i++) {
cin >> a >> b >> c;
if(ma[a][b]) ma[a][b] += c;
else {
arc[a].push(b);
ma[a][b] = c;
}
}

int ret = edmonds(1, n);
cout << ret << endl;
}
return 0;
}
